可配置参数

public class FightServerParameter
{
      public string ServerIP;//后端服务器IP
      public ushort ServerPort;//后端服务器端口
      public string SelfIP;//自身IP，如果空着不填则会在第一次开启时自动获取并填入
      public ushort SelfPort;//管理服务器与本地实例通讯端口
      public ushort[] FightAllotPorts;//本地实例允许使用的端口集合
      public string FightServerPath;//战斗服务器实例路径，如果未填，会自动生成一个路径，但允许更改
      public string UpdateResourcePath;//资源更新路径，暂时没用
      public int PreparaProcessNum = 1;//预备进程数量，一定程度决定载荷，建议设置为服务器总容量的2/3左右
      public (string, string)[] MailUserAndPermission;//邮箱系统用到的账户与SMTP授权码，如果不知如何获取授权码，请自行百度
      public string[] TargetMails;//邮件发送到的目标邮箱地址
}

配置文件所在路径：当前程序运行路径+Resources+目标平台名称+Configuration.txt
